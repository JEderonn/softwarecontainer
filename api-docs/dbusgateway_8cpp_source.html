36 ReturnCode DBusGateway::setConfig(const json_t *config)
37 {
38  try {
39  log_debug() << "Setting config for dbus session bus";
40  ReturnCode sessionBusParseResult = sessionBus.setConfig(config);
41 
42  log_debug() << "Setting config for dbus system bus";
43  ReturnCode systemBusParseResult = systemBus.setConfig(config);
44 
45  if (isError(sessionBusParseResult) && isError(systemBusParseResult)) {

48  }
49  } catch (GatewayParserError &err) {
50  log_error() << "Bad config in one of the gateway configurations";
51  return ReturnCode::FAILURE;
52  }

54  return ReturnCode::SUCCESS;

57 ReturnCode DBusGateway::activate()
58 {

60  if (sessionBus.isConfigured()) {
61  log_debug() << "Activating dbus session bus";
62  sessionBusActivationResult = sessionBus.activate();

64 
65  ReturnCode systemBusActivationResult = ReturnCode::FAILURE;
66  if (systemBus.isConfigured()) {
67  log_debug() << "Activating dbus system bus";
68  sessionBusActivationResult = systemBus.activate();
69  }
70 
71  if (isError(sessionBusActivationResult) && isError(systemBusActivationResult)) {

73  return ReturnCode::FAILURE;

75  return ReturnCode::SUCCESS;
