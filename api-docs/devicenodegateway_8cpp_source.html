47 bool DeviceNodeGateway::activateGateway()
48 {
49  auto devlist = m_logic.getDevList();
50  if (devlist.empty()) {
51  log_info() << "Activate was called when no devices has been configured.";
52  return false;
53  }
54 
55  for (auto &dev : devlist) {
56  log_info() << "Mapping device " << dev.name;
57 
58  std::string devicePathInContainerOnHost = buildPath(getContainer()->rootFS(), dev.name);
59  std::string deviceParent = parentPath(devicePathInContainerOnHost);
60 
61  // Already existing files can't be converted into directories
62  if (existsInFileSystem(deviceParent) && !isDirectory(deviceParent)) {
63  log_error() << "Parent path of " << dev.name << " already exist and is not a directory";
64  return false;
65  }
66 
67  // If the parent directory does not already exist, we create it.
68  if (!isDirectory(deviceParent) && !createDirectory(deviceParent)) {
69  log_error() << "Could not create parent directory for device " << deviceParent;
70  return false;
71  }
72 
73  if (dev.major != -1) {
74  log_error() << "Device path for container on host: " << devicePathInContainerOnHost;
75  if (existsInFileSystem(devicePathInContainerOnHost)) {
76  log_error() << "The device " << dev.name << " already exists";
77  return false;
78  }
79 
80  // mknod dev.name c dev.major dev.minor
81  FunctionJob job(getContainer(), [&] () {
82  if (existsInFileSystem(dev.name)) {
83  log_error() << "Device already exists";
84  return -1;
85  }
86 
87  int err = mknod(dev.name.c_str(), S_IFCHR | dev.mode, makedev(dev.major, dev.minor));
88  if (err != 0) {
89  log_error() << "Error happened while trying to run mknod " << dev.name
90  << " in container: " << err << " - " << strerror(errno);
91  }
92  return err;
93  });
94 
95  job.start();
96  job.wait();
97  if (job.isError()) {
98  log_error() << "Failed to create device " << dev.name;
99  return false;
100  } else {
101  // Make sure the path gets deleted when we destroy this object
102  markFileForDeletion(devicePathInContainerOnHost);
103  }
104  } else {
105  // No major & minor numbers specified => simply map the device from
106  // the host into the container
107  getContainer()->mountDevice(dev.name);
108 
109  if (dev.mode != -1) {
110  FunctionJob job(getContainer(), [&] () {
111  return chmod(dev.name.c_str(), dev.mode);
112  });
113  job.start();
114  job.wait();
115  if (job.isError()) {
116  log_error() << "Could not 'chmod " << dev.mode
117  << "' the mounted device " << dev.name;
118  return false;
119  }
120  }
121  }
122  }
123  return true;
124 }
