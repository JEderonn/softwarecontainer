52 ReturnCode BaseConfigStore::readCapsFromDir(const std::string &dirPath)
53 {
54  if (dirPath.compare("/") == 0) {
55  log_warning() << "Searching for configuration files from root dir not allowed";
56  return ReturnCode::FAILURE;
57  }
58  if (isDirectoryEmpty(dirPath)) {
59  log_warning() << "Path to configuration files is empty: " << dirPath;
60  return ReturnCode::SUCCESS;
61  }
62 
63  std::vector<std::string> files = fileList(dirPath);
64 
65  if (files.empty()) {
66  log_info() << "No configuration files found: " << dirPath;
67  return ReturnCode::SUCCESS;
68  }
69 
70  for (std::string file : files) {
71  std::string filePath = dirPath + file;
72 
73  if (isError(readCapsFromFile(filePath))) {
74  log_warning() << "Could not parse a file in directory: " << filePath;
75  return ReturnCode::FAILURE;
76  }
77  }
78 
79  return ReturnCode::SUCCESS;
80 }

82 ReturnCode BaseConfigStore::readCapsFromFile(const std::string &filePath)
83 {
84  if (!isJsonFile(filePath)) {
85  log_debug() << "File is not a json file: " << filePath;
86  return ReturnCode::FAILURE;
87  }
88 
89  json_error_t error;
90  json_t *fileroot = json_load_file(filePath.c_str(), 0, &error);
91 
92  if (nullptr == fileroot) {
93  log_error() << "Could not parse Service Manifest: "
94  << filePath << ":" << error.line <<" : " << error.text;
95  return ReturnCode::FAILURE;
96  } else if (!json_is_object(fileroot)) {
97  log_error() << "Configuration is not a json object: \n"
98  << filePath << ":" << error.line <<" : " << error.text;
99  return ReturnCode::FAILURE;
100  }
101 
102  json_t *capabilities = json_object_get(fileroot, "capabilities");
103  if (nullptr == capabilities) {
104  log_error() << "Could not parse Capabilities";
105  return ReturnCode::FAILURE;
106  }
107  if (!json_is_array(capabilities)) {
108  log_error() << "Capabilities is not an array";
109  return ReturnCode::FAILURE;
110  }
111 
112  // Can't use json_decref on fileroot, it removes objects too early
113 
114  return parseCapabilities(capabilities);
115 }

117 ReturnCode BaseConfigStore::parseCapabilities(json_t *capabilities)
118 {
119  size_t i;
120  json_t *capability;
121 
122  log_debug() << "Size of capabilities is " << std::to_string(json_array_size(capabilities));
123 
124  json_array_foreach(capabilities, i, capability) {
125  if (!json_is_object(capability)) {
126  log_error() << "Capability is not a json object";
127  return ReturnCode::FAILURE;
128  }
129 
130  std::string capName;
131  if (!JSONParser::read(capability, "name", capName)) {
132  log_error() << "Could not read Capability name";
133  return ReturnCode::FAILURE;
134  }
135 
136  json_t *gateways = json_object_get(capability, "gateways");
137  if (nullptr == gateways) {
138  log_error() << "Could not read Gateways";
139  return ReturnCode::FAILURE;
140  }
141  if (!json_is_array(gateways)) {
142  log_error() << "Gateways is not an array";
143  return ReturnCode::FAILURE;
144  }
145 
146  log_debug() << "Found capability \"" << capName << "\", parsing gateways...";
147 
148  ReturnCode parseGatewayResult = parseGatewayConfigs(capName, gateways);
149  if (isError(parseGatewayResult)) {
150  return ReturnCode::FAILURE;
151  }
152  }
153  return ReturnCode::SUCCESS;
154 }

156 ReturnCode BaseConfigStore::parseGatewayConfigs(std::string capName, json_t *gateways)
157 {
158  if (m_capMap.count(capName) > 0) {
159  log_debug() << "Capability " << capName << " already loaded.";
160  return ReturnCode::SUCCESS;
161  }
162 
163  size_t i;
164  json_t *gateway;
165 
166  GatewayConfiguration gwConf;
167  json_array_foreach(gateways, i, gateway) {
168  if (!json_is_object(gateway)) {
169  log_error() << "Gateway is not a json object";
170  return ReturnCode::FAILURE;
171  }
172 
173  std::string gwID;
174  if (!JSONParser::read(gateway, "id", gwID)) {
175  log_error() << "Could not read Gateway ID";
176  return ReturnCode::FAILURE;
177  }
178 
179  json_t *confs = json_object_get(gateway, "config");
180  if (nullptr == confs) {
181  log_error() << "Could not read Gateway Config";
182  return ReturnCode::FAILURE;
183  }
184  if (!json_is_array(confs)) {
185  log_error() << "Gateway Config is not an array";
186  return ReturnCode::FAILURE;
187  }
188  gwConf.append(gwID, confs);
189  }
190  m_capMap[capName] = gwConf;
191 
192  return ReturnCode::SUCCESS;
193 }

195 std::vector<std::string> BaseConfigStore::fileList(const std::string &dirPath)
196 {
197  std::vector<std::string> files;
198 
199  DIR *dirFile = opendir(dirPath.c_str());
200  if (nullptr == dirFile) {
201  log_error() << "Could not open directory: " << dirPath;
202  return files;
203  }
204 
205  struct dirent *hFile;
206  while ((hFile = readdir(dirFile)) != nullptr) {
207  std::string filename(hFile->d_name);
208  if (isJsonFile(filename)) {
209  files.push_back(filename);
210  }
211  }
212  closedir(dirFile);
213 
214  return files;
215 }
