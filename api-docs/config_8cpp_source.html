82 template<typename T>
83 T Config::getValue(const std::string &group, const std::string &key, const std::map<std::string, T> &options) const
84 {
85  // Check if this key has been explicitly set and if so, return that value
86  typename std::map<std::string, T>::const_iterator position = options.find(key);
87  if (position != options.end()) {
88  // If the end wasn't reached, the value is where the iterator is now
89  log_debug() << "Config \""
90  << group << "::" << key
91  << "\" found in explicitly set options. Value: \""
92  << position->second << "\"";
93  return position->second;
94  }
95 
96  T value;
97 
98  // The value was not explicitly set so look for value in the config
99  if (!m_config->has_group(group)) {
100  // Incorrect group is a configuration error, don't continue to fall back to default
101  log_error() << "Incorrect config group: \"" << group << "\"";
102  throw softwarecontainer::ConfigError();
103  }
104 
105  if (m_config->has_key(group, key)) {
106  try {
107  value = getGlibValue<T>(group, key);
108  log_debug() << "Config \""
109  << group << "::" << key
110  << "\" found in config file. Value: \""
111  << value << "\"";
112  return value;
113  } catch (Glib::Error &error) {
114  log_error() << "Could not parse string value: \"" << error.what() << "\"";
115  throw softwarecontainer::ConfigError();
116  }
117  }
118 
119  // We should always have defaults set, but check just to be sure.
120  if (!m_defaults) {
121  log_error() << "No config defaults set, cannot continue";
122  throw softwarecontainer::ConfigError();
123  }
124 
125  // Value has not been set at all, fall back on default
126  try {
127  value = m_defaults->getValue<T>(key);
128  log_debug() << "Config \""
129  << group << "::" << key
130  << "\" is falling back on default. Value: \""
131  << value << "\"";
132  } catch (std::exception &error) {
133  log_error() << "No config default found for config \"" << group << "::" << key << "\"";
134  throw softwarecontainer::ConfigError();
135  }
136  return value;
137 }
