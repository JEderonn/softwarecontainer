75 DBusHandlerResult filter_cb (DBusConnection *conn,
76  DBusMessage *msg,
77  void *user_data)
78 {
79  /* Data arriving from client */
80  guint32 serial;
81  DBusHandlerResult retval = DBUS_HANDLER_RESULT_HANDLED;
82 
83  /* Handle Hello */
84  if (dbus_message_get_type (msg) ==
85  DBUS_MESSAGE_TYPE_METHOD_CALL &&
86  strcmp (dbus_message_get_path(msg),
87  "/org/freedesktop/DBus") == 0 &&
88  strcmp (dbus_message_get_interface(msg),
89  "org.freedesktop.DBus") == 0 &&
90  strcmp (dbus_message_get_destination(msg),
91  "org.freedesktop.DBus") == 0 &&
92  strcmp (dbus_message_get_member (msg), "Hello") == 0) {
93 
94  DBusMessage *welcome;
95  const gchar *dbus_local_name;
96 
97  dbus_local_name = dbus_bus_get_unique_name (
98  dbus_g_connection_get_connection(master_conn));
99 
100  if (verbose) {
101  g_print("Hello received\n");
102  }
103 
104  welcome = dbus_message_new_method_return (msg);
105  if (!dbus_message_append_args (welcome,
106  DBUS_TYPE_STRING,
107  &dbus_local_name,
108  DBUS_TYPE_INVALID)) {
109  g_printerr("Cannot reply to Hello message\n");
110  exit(1);
111  }
112  dbus_connection_send(conn, welcome, &serial);
113 
114  dbus_message_unref (welcome);
115  goto out;
116  }
117 
118  /* Handle Disconnected */
119  if (dbus_message_get_type(msg) ==
120  DBUS_MESSAGE_TYPE_SIGNAL &&
121  strcmp (dbus_message_get_interface (msg),
122  "org.freedesktop.DBus.Local") == 0 &&
123  strcmp (dbus_message_get_member (msg),
124  "Disconnected") == 0) {
125 
126  /* connection was disconnected */
127  if (verbose) {
128  g_print("connection was disconnected\n");
129  }
130 
131  dbus_connection_close (dbus_conn);
132  dbus_connection_unref (dbus_conn);
133  dbus_conn = NULL;
134  exit(0);
135  goto out;
136  }
137 
138  /* Forward */
139  if (dbus_message_get_interface (msg) == NULL ||
140  strcmp (dbus_message_get_interface(msg),
141  "org.freedesktop.DBus") == 0)
142  {
143  dbus_connection_send(
144  dbus_g_connection_get_connection(master_conn),
145  msg,
146  &serial);
147  } else if (is_allowed("outgoing",
148  dbus_message_get_interface (msg),
149  dbus_message_get_path (msg),
150  dbus_message_get_member (msg)))
151  {
152  g_print ("Accepted call to '%s' from "
153  "client to '%s' on '%s'.\n",
154  dbus_message_get_member (msg),
155  dbus_message_get_interface(msg),
156  dbus_message_get_path (msg));
157 
158  dbus_connection_send (
159  dbus_g_connection_get_connection (master_conn),
160  msg,
161  &serial);
162  } else {
163  g_print ("Rejected call to '%s' from "
164  "client to '%s' on '%s'.\n",
165  dbus_message_get_member (msg),
166  dbus_message_get_interface (msg),
167  dbus_message_get_path (msg));
168  retval = DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
169  }
170 
171 out:
172  return retval;
173 }

239 gboolean match_method(const json_t *rule, const char *comparison)
240 {
241  json_t *json_entry;
242  gchar *string;
243 
244  json_entry = json_object_get(rule, "method");
245  if (json_is_array(json_entry)) {
246  size_t ix;
247  json_t *val;
248 
249  json_array_foreach(json_entry, ix, val) {
250  if (!json_is_string(val)) {
251  if (verbose) {
252  g_print("Entry in method array is not a string.");
253  }
254  return FALSE;
255  }
256 
257  string = (gchar*) json_string_value (val);
258  if ((strcmp (string, "") != 0) && g_pattern_match_simple (string, comparison)) {
259  if (verbose) {
260  g_print("was a match\n");
261  }
262  return TRUE;
263  }
264  }
265  } else if (json_is_string (json_entry)) {
266  string = (gchar*) json_string_value (json_entry);
267  return compare_entry(comparison, string);
268  }
269 
270  return FALSE;
271 }

286 gboolean is_allowed (const char *direction,
287  const char *interface,
288  const char *path,
289  const char *member)
290 {
291  size_t i;
292  json_t *rule;
293  gboolean direction_ok, interface_ok, object_path_ok, method_ok;
294 
295  /* Check all rules until a match is found. When a match is found we
296  don't check any following rules. This means that a more permissive
297  rule will trump less permissive rules. */
298  for (i = 0; i < json_array_size (json_filters); i++) {
299  direction_ok, interface_ok, object_path_ok, method_ok = FALSE;
300 
301  /* Get the JSON array containing the JSON objects */
302  rule = json_array_get (json_filters, i);
303  if (rule == NULL || !json_is_object (rule)) {
304  json_decref (rule);
305  break;
306  }
307 
308  direction_ok = match_rule (rule, "direction", direction);
309  interface_ok = match_rule (rule, "interface", interface);
310  object_path_ok = match_rule (rule, "object-path", path);
311  method_ok = match_method(rule, member);
312 
313  /* All entries matched for the rule */
314  if (direction_ok &&
315  interface_ok &&
316  object_path_ok &&
317  method_ok)
318  {
319  return TRUE;
320  }
321 
322  /*
323  * Since direction seems to be a common source of errors, the
324  * following printout is added as a helper to developer
325  */
326  if (!direction_ok &&
327  interface_ok &&
328  object_path_ok &&
329  method_ok)
330  {
331  g_print("Direction '%s' does not match but "
332  "everything else does\n", direction);
333  }
334  }
335 
336  return FALSE;
337 }

351 DBusHandlerResult master_filter_cb (DBusConnection *conn,
352  DBusMessage *msg,
353  void *user_data)
354 {
355  /* Data arriving from server */
356 
357  guint32 serial;
358  DBusHandlerResult retval = DBUS_HANDLER_RESULT_HANDLED;
359 
360  if (!dbus_conn) {
361  exit(1);
362  }
363  /* Make sure that a new connection does not have a unique name
364  that was previously owned by an eavesdropping connection */
365  if (dbus_message_get_member(msg) != NULL &&
366  strcmp(dbus_message_get_member(msg), "NameAcquired") == 0)
367  {
368  const char *dest = dbus_message_get_destination(msg);
369  if (verbose)
370  {
371  g_print ("NameAcquired received by %s\n", dest);
372  }
373 
374  if (dest != NULL &&
375  is_conn_known_eavesdropper(dest))
376  {
377  if (verbose)
378  {
379  g_print("New connection's unique name ('%s')"
380  " was previously known as an eavesdropper."
381  " Removed old entry...\n", dest);
382  }
383  remove_name_from_known_eavesdroppers(dest);
384  }
385  }
386 
387  /* Forward */
388  if (dbus_message_get_interface(msg) == NULL ||
389  strcmp(dbus_message_get_interface(msg),
390  "org.freedesktop.DBus") == 0)
391  {
392  if (is_incoming_eavesdropping(msg) &&
393  !is_conn_known_eavesdropper(dbus_message_get_sender(msg)))
394  {
395  eavesdropping_conns =
396  g_list_append (eavesdropping_conns,
397  (gpointer) dbus_message_get_sender(msg));
398  }
399 
400  dbus_connection_send(dbus_conn, msg, &serial);
401  } else if (is_conn_known_eavesdropper (dbus_bus_get_unique_name(conn)))
402  {
403  if (verbose) {
404  g_print ("'%s' is an eavesdropping connection, let it go...\n",
405  dbus_bus_get_unique_name(conn));
406  }
407  } else if (is_allowed("incoming",
408  dbus_message_get_interface (msg),
409  dbus_message_get_path (msg),
410  dbus_message_get_member (msg)))
411  {
412  g_print("Accepted call to '%s' from server to '%s' on '%s'.\n",
413  dbus_message_get_member (msg),
414  dbus_message_get_interface (msg),
415  dbus_message_get_path (msg));
416  dbus_connection_send(dbus_conn, msg, &serial);
417  } else {
418  g_print("Rejected call to '%s' from server to '%s' on '%s'.\n",
419  dbus_message_get_member (msg),
420  dbus_message_get_interface (msg),
421  dbus_message_get_path (msg));
422  retval = DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
423  }
424 
425  return retval;
426 }

547 void new_connection_cb (DBusServer *server, DBusConnection *conn, void *data) {
548  pid_t pid;
549  pid_t forked;
550  GError *error = NULL;
551 
552  forked = fork();
553  pid = getpid();
554 
555  if (forked != 0) {
556  if (verbose)
557  g_print ("in main process, pid: %d\n", pid);
558 
559  /* Reconfigure the master socket as forking will break it */
560  start_bus();
561  return;
562  } else {
563  if (verbose)
564  g_print ("in child process, pid: %d\n", pid);
565  }
566 
567  if (master_conn != NULL) {
568  g_print ("master_conn already initialized\n");
569  exit (1);
570  }
571 
572  if (dbus_conn != NULL) {
573  g_print("dbus_conn already initialized\n");
574  exit (1);
575  }
576 
577  /* Init master connection */
578  master_conn = dbus_g_bus_get (bus, &error);
579  if (!master_conn) {
580  g_printerr ("Failed to open connection to bus: %s\n",
581  error->message);
582  g_clear_error (&error);
583  exit(1);
584  }
585 
586  dbus_connection_add_filter (
587  dbus_g_connection_get_connection(master_conn),
588  master_filter_cb,
589  NULL,
590  NULL);
591 
592  if (verbose) {
593  g_print("New connection\n");
594  }
595 
596  dbus_connection_ref (conn);
597  dbus_connection_setup_with_g_main (conn, NULL);
598  dbus_connection_add_filter (conn, filter_cb, NULL, NULL);
599 
600  dbus_connection_set_unix_user_function (conn,
601  allow_all_connections,
602  NULL,
603  NULL);
604 
605  dbus_connection_set_allow_anonymous (conn, TRUE);
606  dbus_conn = conn;
607 }

640 int parse_json_from_stdin (const char *section) {
641  size_t i;
642  json_error_t error;
643  json_t *root, *config, *rule;
644  char *full_section = calloc (sizeof (char), 30);
645  int retval = 0;
646  int config_found = 0;
647 
648  snprintf (full_section, 30, "dbus-gateway-config-%s", section);
649 
650  /* Allow multiple configuration snippets, each as it's own root obj */
651  do {
652 
653  /* Get root JSON object */
654  root = json_loadf (stdin, JSON_DISABLE_EOF_CHECK, &error);
655 
656  if (!root) {
657  /* If no config was found this is an error, otherwise it's normal */
658  if (!config_found) {
659  g_printerr ("error: on line %d: %s\n", error.line, error.text);
660  retval = 1;
661  }
662  break;
663  }
664 
665  /* We have atleast one root object in file */
666  config_found = 1;
667 
668  // Get array
669  config = json_object_get (root, full_section);
670 
671  if (!json_is_array(config)) {
672  g_printerr("error: %s is not present in config, or not an array. "
673  "Fix your config\n", full_section);
674  json_decref (config);
675  retval = 1;
676  break;
677  }
678  if (NULL == json_filters) {
679  json_filters = config;
680  } else {
681  if (0 != json_array_extend(json_filters, config)){
682  g_printerr("Error extending config array\n");
683  }
684  }
685  } while (root);
686 
687  return retval;
688 }

697 int main(int argc, char *argv[]) {
698  GMainLoop *mainloop = NULL;
699  GError *error = NULL;
700 
701  /* Support --version */
702  if (argc == 2 && strcmp(argv[1], "--version") == 0) {
703  print_usage();
704  exit(0);
705  }
706 
707  /* Check for right number of args */
708  if (argc < 3) {
709  print_usage();
710  exit(1);
711  }
712 
713  /* Extract address */
714  address = g_strconcat("unix:path=", argv[1], NULL);
715  if (strcmp (argv[2], "system") == 0) {
716  bus = DBUS_BUS_SYSTEM;
717  } else if (strcmp(argv[2], "session") == 0) {
718  bus = DBUS_BUS_SESSION;
719  } else {
720  g_print ("Must give bus type as second argument (either session or system).\n");
721  exit (1);
722  }
723 
724  /* Parse JSON */
725  if (parse_json_from_stdin (argv[2]) == 1){
726  g_print ("Something wrong with JSON file. Exiting...\n");
727  exit (1);
728  }
729 
730  /* Set set signal handler */
731  struct sigaction sa;
732  sa.sa_handler = &handle_sigchld;
733  sigemptyset(&sa.sa_mask);
734  sa.sa_flags = SA_RESTART | SA_NOCLDSTOP;
735  if (sigaction(SIGCHLD, &sa, 0) == -1) {
736  perror(0);
737  exit(1);
738  }
739 
740  /* Start listening */
741  start_bus ();
742  mainloop = g_main_loop_new (NULL, FALSE);
743  g_main_loop_run (mainloop);
744 
745  return 0;
746 }
