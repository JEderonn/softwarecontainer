29 NetworkGateway::NetworkGateway(const int32_t id,
30  const std::string gateway,
31  const uint8_t maskBits) :
32  Gateway(ID),
33  m_gateway(gateway),
34  m_interfaceInitialized(false),
35  m_containerID(id)
36 {
37  /*
38  * Netmask are used for determining range of ip adress assignment. maskBits represent bit-count
39  * for creating ip range starting from least significant bit of m_gateway. Since an ipv4 address
40  * consist of 32 bits, maskBits shall not be greater than 32. And since bit 0 and bit 31 cannot
41  * give a range but a single exact value, those will not be accepted as a maskBits.
42  */
43  if (maskBits > 31 || maskBits < 1) {
44  log_error() << "inappropriate netmask : " << maskBits;
45  throw ReturnCode::FAILURE;
46  }
47 
48  // value of m_netmask interprets maskBits to a mask integer to calculate range.
49  m_netmask = (1L<<(32-maskBits));
50 }

70 bool NetworkGateway::activateGateway()
71 {
72  if (!hasContainer()) {
73  log_error() << "activate was called on an EnvironmentGateway which has no associated container";
74  return false;
75  }
76 
77  if (m_gateway.size() != 0) {
78  log_debug() << "Default gateway set to " << m_gateway;
79  } else {
80  log_debug() << "No gateway. Network access will be disabled";
81  }
82 
83  if (isError(isBridgeAvailable())) {
84  log_error() << "Bridge not available, expected gateway to be " << m_gateway;
85  return false;
86  }
87 
88  if (isError(generateIP())) {
89  return false;
90  }
91 
92  auto returnValue = up();
93 
94  for (auto entry : m_entries) {
95  FunctionJob job (getContainer(), [&] () {
96  if (isSuccess(entry.applyRules())) {
97  return 0;
98  } else {
99  return 1;
100  }
101  });
102  job.start();
103  int returnCode = job.wait();
104 
105  if (returnCode != 0) {
106  log_debug() << "Failed to apply rules for entry: " << entry.toString();
107  }
108 
109  }
110 
111  return isSuccess(returnValue);
112 }

119 ReturnCode NetworkGateway::generateIP()
120 {
121  log_debug() << "Generating ip-address";
122  // IP generation is designed for Ipv4 in case of transition to IPv6 it should be revised
123  uint32_t internetAddress;
124  if ( 1 != inet_pton(AF_INET, m_gateway.c_str(), &internetAddress)) {
125  log_error() << m_gateway << " does not represent a valid network address";
126  return ReturnCode::FAILURE;
127  }
128 
129  internetAddress = ntohl(internetAddress);
130 
131  if ((internetAddress | m_netmask) < (internetAddress + m_containerID + 1)) {
132  log_error() << "There are no suitable ip address for this container.";
133  return ReturnCode::FAILURE;
134  }
135 
136  internetAddress += (m_containerID + 1);
137  internetAddress = htonl(internetAddress);
138  m_ip.s_addr = internetAddress;
139 
140  // convert ip to human readable form just for debug
141  char convertionTmp[INET_ADDRSTRLEN];
142  inet_ntop(AF_INET, &internetAddress, convertionTmp, INET_ADDRSTRLEN);
143  log_debug() << "IP set to " << convertionTmp;
144 
145  return ReturnCode::SUCCESS;
146 }

161 ReturnCode NetworkGateway::up()
162 {
163  if (m_interfaceInitialized) {
164  log_debug() << "Interface already configured";
165  return ReturnCode::SUCCESS;
166  }
167 
168  log_debug() << "Attempting to bring up eth0";
169  FunctionJob jobBringUpEthernet(getContainer(), [this] {
170  Netlink n;
171 
172  Netlink::LinkInfo iface;
173  if (isError(n.findLink("eth0", iface))) {
174  return 1;
175  }
176 
177  int ifaceIndex = iface.first.ifi_index;
178  if (isError(n.linkUp(ifaceIndex))) {
179  return 2;
180  }
181 
182  return isSuccess(n.setIP(ifaceIndex, m_ip, 24)) ? 0 : 3;
183  });
184 
185  jobBringUpEthernet.start();
186 
187 
188  int returnCode = jobBringUpEthernet.wait();
189  if (returnCode == 1) {
190  log_error() << "Could not find interface eth0 in container";
191  return ReturnCode::FAILURE;
192  }
193 
194  if (returnCode == 2) {
195  log_error() << "Could not bring interface eth0 up in container";
196  return ReturnCode::FAILURE;
197  }
198 
199  if (returnCode == 3) {
200  log_error() << "Could not set IP-address";
201  return ReturnCode::FAILURE;
202  }
203 
204  m_interfaceInitialized = true;
205  return setDefaultGateway();
206 }

208 ReturnCode NetworkGateway::down()
209 {
210  log_debug() << "Attempting to configure eth0 to 'down state'";
211  FunctionJob job(getContainer(), [this] {
212  Netlink n;
213  Netlink::LinkInfo iface;
214  if (isError(n.findLink("eth0", iface))) {
215  log_error() << "Could not find interface eth0 in container";
216  return 1;
217  }
218 
219  if (isError(n.linkDown(iface.first.ifi_index))) {
220  log_error() << "Could not bring interface eth0 down in container";
221  return 2;
222  }
223 
224  return 0;
225  });
226  job.start();
227 
228  int returnCode = job.wait();
229  if (returnCode != 0) {
230  log_error() << "Configuring eth0 to 'down state' failed.";
231  return ReturnCode::FAILURE;
232  }
233 
234  return ReturnCode::SUCCESS;
235 }
