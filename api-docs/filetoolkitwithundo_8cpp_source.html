34 FileToolkitWithUndo::~FileToolkitWithUndo()
35 {
36  bool success = true;
37  // Clean up all created directories, files, and mount points
38 
39  while (!m_cleanupHandlers.empty()) {
40  CleanUpHandler *c = m_cleanupHandlers.back();
41  m_cleanupHandlers.pop_back();

43  if (!c->clean()) {
44  success = false;
45  }
46 
47  delete c;
48  }
49 
50  if(!success) {
51  log_error() << "One or more cleanup handlers returned error status, please check the log";

68 bool FileToolkitWithUndo::createDirectory(const std::string &path)
69 {
70  log_debug() << "createDirectory(" << path << ") called";
71  if (isDirectory(path)) {
72  return true;
73  }
74 
75  if(!createParentDirectory(path)) {
76  log_error() << "Couldn't create parent directory for " << path;
77  return false;
78  }
79 
80  if (mkdir(path.c_str(), S_IRWXU | S_IRWXG | S_IRWXO) == -1) {
81  log_error() << "Could not create directory " << path << ": " << strerror(errno);
82  return false;
83  }
84 
85  if (!pathInList(path)) {
86  m_cleanupHandlers.push_back(new DirectoryCleanUpHandler(path));
87  }
88  log_debug() << "Created directory " << path;
89 
90  return true;
91 }

109 bool FileToolkitWithUndo::bindMount(const std::string &src,
110  const std::string &dst,
111  bool readOnly,
112  bool enableWriteBuffer)
113 {
114  unsigned long flags = MS_BIND;
115  std::string fstype;
116  const void *data = nullptr;
117  int mountRes;
118 
119  if (!existsInFileSystem(src)) {
120  log_error() << src << " does not exist on the host, can not bindMount";
121  return false;
122  }
123 
124  log_debug() << "Bind-mounting " << src << " in " << dst << ", flags: " << flags;
125 
126  if(enableWriteBuffer) {
127  std::string upperDir = tempDir("/tmp/sc-bindmount-upper-XXXXXX");
128  std::string workDir = tempDir("/tmp/sc-bindmount-work-XXXXXX");
129  fstype.assign("overlay");
130 
131  std::ostringstream os;
132  os << "lowerdir=" << src << ",upperdir=" << upperDir << ",workdir=" << workDir;
133  data = os.str().c_str();
134 
135  log_debug() << "enableWriteBuffer, config: " << os.str();
136 
137  mountRes = mount("overlay", dst.c_str(), fstype.c_str(), flags, data);
138  } else {
139  mountRes = mount(src.c_str(), dst.c_str(), fstype.c_str(), flags, data);
140  }
141 
142  if (mountRes == 0) {
143  log_verbose() << "Bind-mounted folder " << src << " in " << dst;
144  m_cleanupHandlers.push_back(new MountCleanUpHandler(dst));
145  } else {
146  log_error() << "Could not mount into container: src=" << src
147  << " , dst=" << dst << " err=" << strerror(errno);
148  return false;
149  }
150 
151  if (readOnly) {
152  flags = MS_REMOUNT | MS_RDONLY | MS_BIND;
153 
154  log_debug() << "Re-mounting read-only" << src << " in "
155  << dst << ", flags: " << flags;
156  mountRes = mount(src.c_str(), dst.c_str(), fstype.c_str(), flags, data);
157  if (mountRes != 0) {
158  // Failure
159  log_error() << "Could not re-mount " << src << " , read-only on "
160  << dst << " err=" << strerror(errno);
161  return false;
162  }
163  }
164  return true;
165 }

167 bool FileToolkitWithUndo::overlayMount(const std::string &lower,
168  const std::string &upper,
169  const std::string &work,
170  const std::string &dst)
171 {
172  std::string fstype = "overlay";
173  unsigned long flags = MS_BIND;
174 
175  if (!createDirectory(lower)
176  || !createDirectory(upper)
177  || !createDirectory(work)
178  || !createDirectory(dst))
179  {
180  log_error() << "Failed to create lower/upper/work directory for overlayMount. lower=" <<
181  lower << ", upper=" << upper << ", work=" << work;
182  return false;
183  }
184 
185  std::string mountoptions = logging::StringBuilder() << "lowerdir=" << lower
186  << ",upperdir=" << upper
187  << ",workdir=" << work;
188 
189  int mountRes = mount("overlay", dst.c_str(), fstype.c_str(), flags, mountoptions.c_str());
190 
191  if (mountRes == 0) {
192  log_verbose() << "overlayMounted folder " << lower << " in " << dst;
193  m_cleanupHandlers.push_back(new MountCleanUpHandler(dst));
194  if (!pathInList(upper)) {
195  m_cleanupHandlers.push_back(new DirectoryCleanUpHandler(upper));
196  }
197  m_cleanupHandlers.push_back(new OverlaySyncCleanupHandler(upper, lower));
198  if (!pathInList(work)) {
199  m_cleanupHandlers.push_back(new DirectoryCleanUpHandler(work));
200  }
201  } else {
202  log_error() << "Could not mount into container: upper=" << upper
203  << ",lower=" << lower
204  << ",work=" << work
205  << " at dst=" << dst << " err=" << strerror(errno);
206  return false;
207  }
208 
209  return true;
210 }

218 bool FileToolkitWithUndo::createSharedMountPoint(const std::string &path)
219 {
220  auto mountRes = mount(path.c_str(), path.c_str(), "", MS_BIND, nullptr);
221  if (mountRes != 0) {
222  log_error() << "Could not bind mount " << path << " to itself";
223  return false;
224  }
225 
226  mountRes = mount(path.c_str(), path.c_str(), "", MS_UNBINDABLE, nullptr);
227  if (mountRes != 0) {
228  log_error() << "Could not make " << path << " unbindable";
229  return false;
230  }
231 
232  mountRes = mount(path.c_str(), path.c_str(), "", MS_SHARED, nullptr);
233  if (mountRes != 0) {
234  log_error() << "Could not make " << path << " shared";
235  return false;
236  }
237 
238  m_cleanupHandlers.push_back(new MountCleanUpHandler(path));
239  log_debug() << "Created shared mount point at " << path;
240 
241  return true;
242 }
