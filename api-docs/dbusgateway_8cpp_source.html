34 ReturnCode DBusGateway::setConfig(const std::string &config)
35 {
36  try {
37  log_debug() << "Setting config for dbus session bus";
38  ReturnCode sessionBusParseResult = sessionBus.setConfig(config);
39 
40  log_debug() << "Setting config for dbus system bus";

42

48  log_error() << "Bad config in one of the gateway configurations";
49  return ReturnCode::FAILURE;

55 ReturnCode DBusGateway::activate()

58  if (sessionBus.isConfigured()) {
59  log_debug() << "Activating dbus session bus";
60  sessionBusActivationResult = sessionBus.activate();

62 
63  ReturnCode systemBusActivationResult = ReturnCode::FAILURE;
64  if (systemBus.isConfigured()) {
65  log_debug() << "Activating dbus system bus";
66  sessionBusActivationResult = systemBus.activate();

68 
69  if (isError(sessionBusActivationResult) && isError(systemBusActivationResult)) {

71  return ReturnCode::FAILURE;
