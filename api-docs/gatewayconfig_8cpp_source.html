42 ReturnCode GatewayConfiguration::append(const std::string &id, const std::string &jsonConf)
43 {
44  json_error_t jsonError;
45  size_t flags = 0; // default flags
46 
47  json_t *json = json_loads(jsonConf.c_str(), flags, &jsonError);
48  if (nullptr == json) {
49  log_error() << "Could not parse given string to JSON. Due to: " << jsonError.text << " at line: " << jsonError.line;
50  return ReturnCode::FAILURE;
51  }
52 
53  auto ret = append(id, json);
54  json_decref(json);
55  return ret;
56 }

58 ReturnCode GatewayConfiguration::append(const std::string &id, json_t *sourceArray)
59 {
60  auto search = m_configMap.find(id);
61  if (search != m_configMap.end()) {
62  // Get the arrays
63  json_t *destArray = search->second;
64  json_t *backupArray = json_deep_copy(destArray);
65 
66  // Add all elements of the source array to the destination array
67  size_t index;
68  json_t *value;
69  json_array_foreach(sourceArray, index, value) {
70  json_t *copy = json_deep_copy(value);
71  if (json_array_append(destArray, copy) < 0) {
72  log_error() << "Could not add Gateway Config to json array: " << id;
73  m_configMap[id] = backupArray;
74  json_decref(destArray);
75  return ReturnCode::FAILURE;
76  }
77  }
78  json_decref(backupArray);
79  } else {
80  m_configMap[id] = sourceArray;
81  json_incref(sourceArray);
82  }
83 
84  return ReturnCode::SUCCESS;
85 }
