137 MandatoryConfigs ConfigDefinition::mandatory()
138 {
139  MandatoryConfigs mandatoryConfigs = MandatoryConfigs();
140 
141  for (auto config : CONFIGS) {
142  std::string group = std::get<ITEM_INDEX_GROUP>(config);
143  std::string key = std::get<ITEM_INDEX_KEY>(config);
144  MandatoryFlag mandatoryFlag = std::get<ITEM_INDEX_MANDATORY>(config);
145 
146  if (Mandatory == mandatoryFlag) {
147  mandatoryConfigs.push_back(UniqueKey(group, key));
148  }
149  }
150 
151  return mandatoryConfigs;
152 }

154 TypeMap ConfigDefinition::typeMap()
