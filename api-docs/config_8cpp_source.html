78  /* Checking for mandatory config group is only relevant for the main config source
79  */
80  for (std::string group : mandatory.groups()) {
81  log_debug() << "Checking for mandatory config group: \"" << group << "\"";
82  if (!m_config->has_group(group)) {
83  log_error() << "Missing mandatory config group: \"" + group + "\"";
84  throw ConfigMandatoryError("Missing mandatory config group: \"" + group + "\"");
85  }
86  }
87 
88  /* Check that each mandatory config is present in one of the sources using the normal
89  * retrieval methods in order to preserve the priority order.
90  */
91  for (std::tuple<std::string, std::string, ConfigType> config : mandatory.configs()) {
92  std::string group = std::get<0>(config);
93  std::string key = std::get<1>(config);
94  ConfigType type = std::get<2>(config);
95 
96  log_debug() << "Checking for presence of mandatory config: \"" << key << "\"";
97  /* TODO: The below only checks for presence, it should also add the values to something
98  * that can be used for retrieval later. */
99  try {
100  switch (type) {
101  case ConfigType::String:
102  getStringValue(group, key);
103  break;
104  case ConfigType::Integer:
105  getIntegerValue(group, key);
106  break;
107  case ConfigType::Boolean:
108  getBooleanValue(group, key);
109  break;
110  default:
111  break;
112  }
113  } catch (ConfigError &error) {
114  log_error() << "Missing mandatory config: \"" << key << "\"";
115  throw ConfigMandatoryError("Missing mandatory config: \"" + key + "\"");
116  }
117  }

145 template<typename T>
146 T Config::getValue(const std::string &group, const std::string &key, const std::map<std::string, T> &options) const
147 {
148  // Check if this key has been explicitly set and if so, return that value
149  typename std::map<std::string, T>::const_iterator position = options.find(key);
150  if (position != options.end()) {
151  // If the end wasn't reached, the value is where the iterator is now
152  log_debug() << "Config \""
153  << group << "::" << key
154  << "\" found in explicitly set options. Value: \""
155  << position->second << "\"";
156  return position->second;
157  }
158 
159  T value;
160 
161  // The value was not explicitly set so look for value in the config
162  if (!m_config->has_group(group)) {
163  // Incorrect group is a configuration error, don't continue to fall back to default
164  log_error() << "Incorrect config group: \"" << group << "\"";
165  throw ConfigError();
166  }
167 
168  if (m_config->has_key(group, key)) {
169  try {
170  value = getGlibValue<T>(group, key);
171  log_debug() << "Config \""
172  << group << "::" << key
173  << "\" found in config file. Value: \""
174  << value << "\"";
175  return value;
176  } catch (Glib::Error &error) {
177  log_error() << "Could not parse string value: \"" << error.what() << "\"";
178  throw ConfigError();
179  }
180  }
181 
182  // We should always have defaults set, but check just to be sure.
183  if (!m_defaults) {
184  log_error() << "No config defaults set, cannot continue";
185  throw ConfigError();
186  }
187 
188  // Value has not been set at all, fall back on default
189  try {
190  value = m_defaults->getValue<T>(key);
191  log_debug() << "Config \""
192  << group << "::" << key
193  << "\" is falling back on default. Value: \""
194  << value << "\"";
195  } catch (std::exception &error) {
196  log_error() << "No config default found for config \"" << group << "::" << key << "\"";
197  throw ConfigError();
198  }
199 
200  return value;
201 }
