45 bool DeviceNodeGateway::activateGateway()
46 {
47  auto devlist = m_logic.getDevList();
48  if (devlist.empty()) {
49  log_info() << "Activate was called when no devices has been configured.";
50  return false;
51  }
52 
53  for (auto &dev : devlist) {
54  log_info() << "Mapping device " << dev.name;
55 
56  std::string devicePathInContainerOnHost = buildPath(getContainer()->rootFS(), dev.name);
57  std::string deviceParent = parentPath(devicePathInContainerOnHost);
58 
59  // Already existing files can't be converted into directories
60  if (existsInFileSystem(deviceParent) && !isDirectory(deviceParent)) {
61  log_error() << "Parent path of " << dev.name << " already exist and is not a directory";
62  return false;
63  }
64 
65  // If the parent directory does not already exist, we create it.
66  if (!isDirectory(deviceParent) && !createDirectory(deviceParent)) {
67  log_error() << "Could not create parent directory for device " << deviceParent;
68  return false;
69  }
70 
71  // Mount device in container
72  getContainer()->mountDevice(dev.name);
73 
74  // If mode is specified, try to set mode for the mounted device.
75  if (dev.mode != -1) {
76  FunctionJob job(getContainer(), [&] () {
77  return chmod(dev.name.c_str(), dev.mode);
78  });
79  job.start();
80  job.wait();
81  if (job.isError()) {
82  log_error() << "Could not 'chmod " << dev.mode
83  << "' the mounted device " << dev.name;
84  return false;
85  }
86  }
87  }
88  return true;
89 }
