24 ReturnCode Gateway::setConfig(const std::string &config)
25 {
26  if (m_state == GatewayState::ACTIVATED) {
27  log_error() << "Can not configure a gateway that is already activated: " << id();
28  throw GatewayError("Gateway already activated");
29  }
30 
31  json_error_t error;
32  json_t *root = json_loads(config.c_str(), 0, &error);
33  if (!root) {
34  std::string errorText = logging::StringBuilder()
35  << "Could not parse config: " << error.text;
36  setConfigRollback(errorText,root);
37  return ReturnCode::FAILURE;
38  }
39 
40  if (!json_is_array(root)) {
41  setConfigRollback("Root JSON element is not an array",root);
42  return ReturnCode::FAILURE;
43  }
44 
45  if (json_array_size(root) == 0) {
46  setConfigRollback("Root JSON array is empty",root);
47  return ReturnCode::FAILURE;
48  }
49 
50  for(size_t i = 0; i < json_array_size(root); i++) {
51  json_t *element = json_array_get(root, i);
52  if (!json_is_object(element)) {
53  setConfigRollback("json configuration is not an object",root);
54  return ReturnCode::FAILURE;
55  }
56 
57  if (isError(readConfigElement(element))) {
58  setConfigRollback("Could not read config element",root);
59  return ReturnCode::FAILURE;
60  }
61  }
62 
63  json_decref(root);
64  m_state = GatewayState::CONFIGURED;
65  return ReturnCode::SUCCESS;
66 }

74 ReturnCode Gateway::activate() {
75  if (m_state == GatewayState::ACTIVATED) {
76  log_error() << "Activate was called on a gateway which was already activated: " << id();
77  throw GatewayError("Gateway already activated");
78  }
79 
80  if (m_state != GatewayState::CONFIGURED) {
81  log_error() << "Activate was called on a gateway which is not in configured state: " << id();
82  throw GatewayError("Gateway is not configured");
83  }
84 
85  if (!hasContainer()) {
86  log_error() << "Activate was called on a gateway which has no associated container: " << id();
87  throw GatewayError("Gateway does not have a container instance");
88  }
89 
90  if (!activateGateway()) {
91  log_error() << "Couldn't activate gateway: " << id();
92  return ReturnCode::FAILURE;
93  }
94 
95  m_state = GatewayState::ACTIVATED;
96  return ReturnCode::SUCCESS;
97 }
