23 bool Gateway::setConfig(const std::string &config)
24 {
25  if (m_state == GatewayState::ACTIVATED) {
26  log_error() << "Can't configure a gateway that is already activated: " << id();
27  }
28 
29  json_error_t error;
30  json_t *root = json_loads(config.c_str(), 0, &error);
31  if (!root) {
32  log_error() << "Could not parse config: " << error.text;
33  return false;
34  }
35 
36  if (!json_is_array(root)) {
37  log_error() << "Root JSON element is not an array";
38  return false;
39  }
40 
41  if (json_array_size(root) == 0) {
42  log_error() << "Root JSON array is empty";
43  return false;
44  }
45 
46  for(size_t i = 0; i < json_array_size(root); i++) {
47  json_t *element = json_array_get(root, i);
48  if (!json_is_object(element)) {
49  log_error() << "json configuration is not an object";
50  return false;
51  }
52 
53  if (isError(readConfigElement(element))) {
54  log_error() << "Could not read config element";
55  return false;
56  }
57  }
58 
59  m_state = GatewayState::CONFIGURED;
60  return true;
61 }

63 bool Gateway::activate() {
64  if (m_state == GatewayState::ACTIVATED) {
65  log_warning() << "Activate was called on a gateway which was already activated: " << id();
66  return false;
67  }
68 
69  if (m_state != GatewayState::CONFIGURED) {
70  log_warning() << "Activate was called on a gateway which is not in configured state: " << id();
71  return false;
72  }
73 
74  if (!hasContainer()) {
75  log_warning() << "Activate was called on a gateway which has no associated container: " << id();
76  return false;
77  }
78 
79  if (!activateGateway()) {
80  log_error() << "Couldn't activate gateway: " << id();
81  return false;
82  }
83 
84  m_state = GatewayState::ACTIVATED;
85  return true;
