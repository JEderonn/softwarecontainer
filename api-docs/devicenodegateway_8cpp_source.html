56 ReturnCode DeviceNodeGateway::applySettings()
57 {
58  auto devlist = m_logic.getDevList();
59  if (devlist.empty()) {
60  log_info() << "Activate was called when no devices has been configured.";
61  return ReturnCode::FAILURE;
62  }
63 
64  for (auto &dev : devlist) {
65  log_info() << "Mapping device " << dev.name;
66 
67  if (dev.major != -1) {
68 
69  // mknod dev.name c dev.major dev.minor
70  pid_t pid = INVALID_PID;
71  getContainer()->executeInContainer([&] () {
72  auto err = mknod(dev.name.c_str(), S_IFCHR | dev.mode,
73  makedev(dev.major, dev.minor));
74  if (err) {
75  log_error() << "Err happened while trying to run mknod in container" <<
76  " error: " << err << " - " << strerror(errno);
77  }
78  return err;
79  }, &pid);
80 
81  if (waitForProcessTermination(pid) != 0) {
82  log_error() << "Failed to create device " << dev.name;
83  return ReturnCode::FAILURE;
84  }
85  } else {
86  // No major & minor numbers specified => simply map the device from
87  // the host into the container
88  getContainer()->mountDevice(dev.name);
89 
90  if (dev.mode != -1) {
91  pid_t pid = INVALID_PID;
92  getContainer()->executeInContainer([&] () {
93  return chmod(dev.name.c_str(), dev.mode);
94  }, &pid);
95 
96  if (waitForProcessTermination(pid) != 0) {
97  log_error() << "Could not 'chmod " << dev.mode
98  << "' the mounted device " << dev.name;
99  return ReturnCode::FAILURE;
100  }
101  }
102  }
103  }
104  return ReturnCode::SUCCESS;
105 }
