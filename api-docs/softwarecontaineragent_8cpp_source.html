99 void SoftwareContainerAgent::removeOldContainers()
100 {
101  char **containerNames = nullptr;
102  struct lxc_container **containerList = nullptr;
103 
104  const char *basePath = lxc_get_global_config_item("lxc.lxcpath");
105  auto num = list_all_containers(basePath, &containerNames, &containerList);
106 
107  if (0 == num) {
108  // there are no container residue. No need to run more.
109  delete containerList;
110  delete containerNames;
111  return;
112  } else if (-1 == num) {
113  log_error() << "An error is occurred while trying to get deprecated container list";
114  throw SoftwareContainerAgentError("An error is occurred while trying to get container list");
115  }
116 
117  log_warning() << num << " unused deprecated containers found";
118  for (auto i = 0; i < num; i++) {
119  struct lxc_container *container = containerList[i];
120  log_debug() << "Deprecated container named " << containerNames[i] << " will be deleted";
121 
122  if (container->is_running(container)) {
123  bool success = container->stop(container);
124  if (!success) {
125  std::string errorMsg = "Unable to stop deprecated container " +
126  std::string(containerNames[i]);
127  throw SoftwareContainerAgentError(errorMsg);
128  }
129  }
130 
131  bool success = container->destroy(container);
132  if (!success) {
133  std::string errorMsg = "Unable to destroy deprecated container " +
134  std::string(containerNames[i]);
135  throw SoftwareContainerAgentError(errorMsg);
136  }
137 
138  log_debug() << "Deprecated container " << containerNames[i] << " is successfully destroyed";
139  delete container;
140  delete containerNames[i];
141  }
142  delete containerList;
143  delete containerNames;
144 }

147 void SoftwareContainerAgent::assertContainerExists(ContainerID containerID)
148 {
149  if (containerID >= INT32_MAX) {
150  std::string errorMessage("Invalid Container ID: "
151  + std::to_string(containerID)
152  + ". ID can not be greater than INT32");
153  log_error() << errorMessage;
154  throw SoftwareContainerError(errorMessage);
155  }
156 
157  if (containerID < 0) {
158  std::string errorMessage("Invalid Container ID: "
159  + std::to_string(containerID)
160  + ". ID can not be negative");
161  log_error() << errorMessage;
162  throw SoftwareContainerError(errorMessage);
163  }
164  if (1 > m_containers.count(containerID)) {
165  std::string errorMessage("Invalid Container ID: "
166  + std::to_string(containerID)
167  + ". No container matching that ID exists.");
168  log_error() << errorMessage;
169  throw SoftwareContainerError(errorMessage);
170  }
171 }

229 pid_t SoftwareContainerAgent::execute(ContainerID containerID,
230  const std::string &cmdLine,
231  const std::string &workingDirectory,
232  const std::string &outputFile,
233  const EnvironmentVariables &env,
234  std::function<void (pid_t, int)> listener)
235 {
236  profilefunction("executeFunction");
237  SoftwareContainerPtr container = getContainer(containerID);
238 
239  /*
240  * We want to always apply any default capabilities we have. The only way to configure
241  * the gateways from the agent is through setCapabilities - which sets the default caps also.
242  *
243  * If it has not been set previously, then we use a call without arguments to setCapabilities
244  * to set it up, since we then should get only the default ones.
245  */
246  if (!m_containers[containerID]->previouslyConfigured()) {
247  log_info() << "Container not configured yet, configuring with default capabilities, if any";
248  GatewayConfiguration gatewayConfigs = m_defaultConfigStore->configs();
249  if (!updateGatewayConfigs(containerID, gatewayConfigs)) {
250  std::string errorMessage("Could not set default capabilities on container " + std::to_string(containerID));
251  log_error() << errorMessage;
252  throw SoftwareContainerError(errorMessage);
253  }
254 
255  }
256 
257  // Set up a CommandJob for this run in the container
258  auto job = container->createCommandJob(cmdLine);
259  job->setOutputFile(outputFile);
260  job->setEnvironmentVariables(env);
261  job->setWorkingDirectory(workingDirectory);
262 
263  // Start it
264  if (isError(job->start())) {
265  std::string errorMessage("Could not start job");
266  log_error() << errorMessage;
267  throw SoftwareContainerError(errorMessage);
268  }
269 
270  // If things went well, do what we need when it exits
271  addProcessListener(m_connections, job->pid(), [listener](pid_t pid, int exitCode) {
272  listener(pid, exitCode);
273  }, m_mainLoopContext);
274 
275  profilepoint("executeEnd");
276 
277  return job->pid();
278 }

343 void SoftwareContainerAgent::setCapabilities(const ContainerID &containerID,
344  const std::vector<std::string> &capabilities)
345 {
346  if (capabilities.empty()) {
347  log_warning() << "Got empty list of capabilities";
348  return;
349  }
350 
351  GatewayConfiguration gatewayConfigs = m_defaultConfigStore->configs();
352  GatewayConfiguration filteredConfigs = m_filteredConfigStore->configsByID(capabilities);
353 
354  // If we get an empty config the user passed a non existent cap name
355  if (filteredConfigs.empty()) {
356  std::string errorMessage("One or more capabilities were not found");
357  log_error() << errorMessage;
358  throw SoftwareContainerError(errorMessage);
359  }
360 
361  gatewayConfigs.append(filteredConfigs);
362 
363  // Update container gateway configuration
364  if (!updateGatewayConfigs(containerID, gatewayConfigs)) {
365  std::string errorMessage("Could not set gateway configuration for capability");
366  log_error() << errorMessage;
367  throw SoftwareContainerError(errorMessage);
368  }
369 }
