31 ReturnCode DeviceNodeGateway::readConfigElement(const json_t *element)
32 {
33  DeviceNodeGateway::Device dev;
34 
35  if (!JSONParser::read(element, "name", dev.name)) {
36  log_error() << "Key \"name\" missing or not a string in json configuration";
37  return ReturnCode::FAILURE;
38  }
39 
40  JSONParser::read(element, "major", dev.major);
41  JSONParser::read(element, "minor", dev.minor);
42  JSONParser::read(element, "mode", dev.mode);
43 
44  const bool majorSpecified = dev.major.length() > 0;
45  const bool minorSpecified = dev.minor.length() > 0;
46 
47  if (majorSpecified | minorSpecified) {
48  if (!checkIsStrValid(dev.major, "Major version", "Major version must be specified when minor is.")
49  || !checkIsStrValid(dev.minor, "Minor version", "Minor version must be specified when major is.")
50  || !checkIsStrValid(dev.mode, "Mode", "Mode has to be specified when major and minor is specified.")) {
51  return ReturnCode::FAILURE;
52  }
53  }
54 
55  m_devList.push_back(dev);
56  return ReturnCode::SUCCESS;
57 }

59 bool DeviceNodeGateway::checkIsStrValid(std::string intStr, std::string name, std::string notSpecified)
60 {
61  if (intStr.length() == 0) {

65 
66  int i = 0;
67  if (!parseInt(intStr.c_str(), &i)) {
68  log_error() << name << " must be represented as an integer.";
69  return false;
70  }

72  return true;

75 bool DeviceNodeGateway::activateGateway()
76 {
77  if (m_devList.empty()) {
78  log_info() << "Activate was called when no devices has been configured.";
79  return false;
80  }
81 
82  for (auto &dev : m_devList) {
83  log_info() << "Mapping device " << dev.name;
84 
85  if (dev.major.length() != 0) {
86 
87  int majorVersion, minorVersion, mode;
88  parseInt(dev.major.c_str(), &majorVersion);
89  parseInt(dev.minor.c_str(), &minorVersion);
90  parseInt(dev.mode.c_str(), &mode);
91 
92  // mknod dev.name c dev.major dev.minor
93  pid_t pid = INVALID_PID;
94  getContainer()->executeInContainer([&] () {
95  return mknod(dev.name.c_str(), S_IFCHR | mode,
96  makedev(majorVersion, minorVersion));
97  }, &pid);
98 
99  if (waitForProcessTermination(pid) != 0) {
100  log_error() << "Failed to create device " << dev.name;
101  return false;
102  }
103  } else {
104  // No major & minor numbers specified => simply map the device from
105  // the host into the container
106  getContainer()->mountDevice(dev.name);
107 
108  if (dev.mode.length() != 0) {
109  const int mode = std::atoi(dev.mode.c_str());
110 
111  pid_t pid = INVALID_PID;
112  getContainer()->executeInContainer([&] () {
113  return chmod(dev.name.c_str(), mode);
114  }, &pid);
115 
116  if (waitForProcessTermination(pid) != 0) {
117  log_error() << "Could not 'chmod " << dev.mode
118  << "' the mounted device " << dev.name;
119  return false;
120  }
121  }
122  }
123  }
124 
125  return true;
126 }
