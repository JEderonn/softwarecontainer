47 /*
48  * Tries to register the agent on the given bus. If the system bus is given and fails,
49  * this tries to register to the session bus instead.
50  *
51  * Throws a ReturnCode on failure
52  */
53 DBus::Connection getBusConnection(bool useSessionBus)
54 {
55  static constexpr const char *AGENT_BUS_NAME = "com.pelagicore.SoftwareContainerAgent";
56  std::string busStr = useSessionBus ? "session" : "system";
57  try {
58  DBus::Connection bus = useSessionBus ? DBus::Connection::SessionBus()
59  : DBus::Connection::SystemBus();
60 
61  // Check if the name is available or not
62  bool alreadyTaken = bus.has_name(AGENT_BUS_NAME);
63  if (alreadyTaken) {
64  log_error() << AGENT_BUS_NAME << " is already taken on the " << busStr << " bus.";
65  throw ReturnCode::FAILURE;
66  }
67 
68  DBus::Connection connection(bus);
69  connection.request_name(AGENT_BUS_NAME);
70  log_debug() << "Registered " << AGENT_BUS_NAME << " on the " << busStr << " bus.";
71  return connection;
72 
73  } catch (DBus::Error &err) {
74  log_warning() << "Could not register " << AGENT_BUS_NAME << " to the " << busStr << " bus";
75  throw ReturnCode::FAILURE;
76  }
77 }
