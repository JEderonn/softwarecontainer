52 /*
53  * Tries to register the agent on the given bus. If the system bus is given and fails,
54  * this tries to register to the session bus instead.
55  *
56  * Throws a ReturnCode on failure
57  */
58 DBus::Connection getBusConnection(bool useSystemBus)
59 {
60  std::string busStr = useSystemBus ? "system" : "session";
61  try {
62  DBus::Connection bus = useSystemBus ? DBus::Connection::SystemBus()
63  : DBus::Connection::SessionBus();
64 
65  // Check if the name is available or not
66  bool alreadyTaken = bus.has_name(AGENT_BUS_NAME);
67  if (alreadyTaken) {
68  log_error() << AGENT_BUS_NAME << " is already taken on the " << busStr << " bus.";
69  throw ReturnCode::FAILURE;
70  }
71 
72  DBus::Connection connection(bus);
73  connection.request_name(AGENT_BUS_NAME);
74  return connection;
75 
76  } catch (DBus::Error &err) {
77  log_warning() << "Could not register " << AGENT_BUS_NAME << " to the " << busStr << " bus";
78  throw ReturnCode::FAILURE;
79  }
80 }

83 int main(int argc, char **argv)
84 {
85  static struct option long_options[] =
86  {
87  { "preload", required_argument, 0, 'p' },
88  { "user", required_argument, 0, 'u' },
89  { "shutdown", required_argument, 0, 's' },
90  { "timeout", required_argument, 0, 't' },
91  { "manifest", required_argument, 0, 'm' },
92  { "session-bus", no_argument, 0, 'b' },
93  { "help", no_argument, 0, 'h' },
94  { 0, 0, 0, 0 }
95  };
96 
97  int preloadCount = 0;
98  int userID = 0;
99  bool shutdownContainers = true;
100  int timeout = 2;
101  std::string servicemanifest = "";
102  bool useSystemBus = true;
103 
104  int option_index = 0;
105  int c = 0;
106  while((c = getopt_long(argc, argv, "p:u:s:t:m:hb", long_options, &option_index)) != -1) {
107  switch(c) {
108  case 'p':
109  if (!parseInt(optarg, &preloadCount)) {
110  usage(argv[0]);
111  exit(1);
112  }
113  break;
114  case 'u':
115  if (!parseInt(optarg, &userID)) {
116  usage(argv[0]);
117  exit(1);
118  }
119  break;
120  case 's':
121  shutdownContainers = std::string(optarg).compare("true") == 0;
122  break;
123  case 't':
124  if (!parseInt(optarg, &timeout)) {
125  usage(argv[0]);
126  exit(1);
127  }
128  break;
129  case 'm':
130  servicemanifest = std::string(optarg);
131  break;
132  case 'b':
133  useSystemBus = false;
134  break;
135  case 'h':
136  usage(argv[0]);
137  exit(0);
138  break;
139 
140  case '?':
141  usage(argv[0]);
142  exit(1);
143  break;
144  }
145  }
146 
147  profilepoint("softwareContainerStart");
148  log_debug() << "Starting softwarecontainer agent";
149 
150  Glib::RefPtr<Glib::MainContext> mainContext = Glib::MainContext::get_default();
151  Glib::RefPtr<Glib::MainLoop> ml = Glib::MainLoop::create(mainContext);
152 
153  DBus::Glib::BusDispatcher dbusDispatcher;
154  DBus::default_dispatcher = &dbusDispatcher;
155  dbusDispatcher.attach(mainContext->gobj());
156 
157  try {
158  DBus::Connection connection = getBusConnection(useSystemBus);
159  SoftwareContainerAgent agent(mainContext,
160  preloadCount,
161  shutdownContainers,
162  timeout,
163  servicemanifest);
164  std::unique_ptr<SoftwareContainerAgentAdaptor> adaptor =
165  std::unique_ptr<SoftwareContainerAgentAdaptor>(
166  new DBusCppAdaptor(connection, AGENT_OBJECT_PATH, agent)
167  );
168 
169  // Register UNIX signal handler
170  g_unix_signal_add(SIGINT, &signalHandler, &ml);
171  g_unix_signal_add(SIGTERM, &signalHandler, &ml);
172  ml->run();
173 
174  log_debug() << "Exiting softwarecontainer agent";
175  return 0;
176  } catch (ReturnCode failure) {
177  log_error() << "Agent initialization failed";
178  return 1;
179  }
