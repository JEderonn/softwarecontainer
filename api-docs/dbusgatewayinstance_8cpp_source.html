76 bool DBusGatewayInstance::activateGateway()
77 {
78  // set DBUS_{SESSION,SYSTEM}_BUS_ADDRESS env variable
79  std::string variable = std::string("DBUS_")
80  + (m_type == SessionProxy ? "SESSION" : "SYSTEM")
81  + std::string("_BUS_ADDRESS");
82  std::string value = "unix:path=" + buildPath("/gateways", socketName());
83  getContainer()->setEnvironmentVariable(variable, value);
84 
85  std::vector<std::string> commandVec = { "dbus-proxy"
86  , m_socket
87  , m_type == SessionProxy ? "session" : "system" };
88 
89  // Give the dbus-proxy access to the real dbus bus address.
90  std::vector<std::string> envVec;
91 
92  bool hasEnvVar = false;
93  std::string envValue = Glib::getenv(variable, hasEnvVar);
94 
95  if (!hasEnvVar) {
96  if (m_type == SessionProxy) {
97  log_error() << "Using DBus gateway in session mode"
98  << " and no " + variable + " set in host environment, dbus-proxy won't work";
99  return false;
100  } else {
101  log_warn() << "Using DBus gateway in system mode"
102  << " and no " + variable + " set in host environment, this could be a problem";
103  }
104  } else {
105  envVec.push_back(variable + "=" + envValue);
106  }
107 
108  if (!startDBusProxy(commandVec, envVec)) {
109  return false;
110  }
111 
112  // Dump to string and clean up
113  char *config_c = json_dumps(m_entireConfig, JSON_COMPACT);
114  std::string config = std::string(config_c);
115  free(config_c);
116 
117  return testDBusConnection(config);
118 }

120 bool DBusGatewayInstance::testDBusConnection(const std::string &config)
121 {
122  log_debug() << "Sending config " << config;
123  unsigned int count = sizeof(char) * config.length();
124  ssize_t written = write(m_infp, config.c_str(), count);
125 
126  // writing didn't work at all
127  if (written == -1) {
128  log_error() << "Failed to write to STDIN of dbus-proxy: " << strerror(errno);
129  return false;
130  } else if (written == (ssize_t)count) {
131  // writing has written exact amout of bytes
132  close(m_infp);
133  m_infp = INVALID_FD;
134  // dbus-proxy might take some time to create the bus socket
135  if (isSocketCreated()) {
136  log_debug() << "Found D-Bus socket: " << m_socket;
137  return true;
138  } else {
139  log_error() << "Did not find any D-Bus socket: " << m_socket;
140  return false;
141  }
142  } else {
143  // something went wrong during the write
144  log_error() << "Failed to write to STDIN of dbus-proxy!";
145  return false;
146  }
147 }

149 bool DBusGatewayInstance::startDBusProxy(const std::vector<std::string> &commandVec,
150  const std::vector<std::string> &envVec)
151 {
152  // Spawn dbus-proxy with access to its stdin.
153  try {
154  Glib::spawn_async_with_pipes(".",
155  commandVec,
156  envVec,
157  Glib::SPAWN_STDOUT_TO_DEV_NULL // Redirect stdout
158  | Glib::SPAWN_STDERR_TO_DEV_NULL // Redirect stderr
159  | Glib::SPAWN_SEARCH_PATH // Search $PATH
160  | Glib::SPAWN_DO_NOT_REAP_CHILD, // Lets us do waitpid
161  sigc::slot<void>(), // child setup
162  &m_pid,
163  &m_infp // stdin
164  );
165  } catch (const Glib::Error &ex) {
166  log_error() << "Failed to launch dbus-proxy";
167  return false;
168  }
169 
170  log_debug() << "Started dbus-proxy: " << m_pid;
171 
172  return true;
173 }

175 bool DBusGatewayInstance::isSocketCreated() const
176 {
177  int maxCount = 1000;
178  int count = 0;
179  do {
180  if (count >= maxCount) {
181  return false;
182  }

189 bool DBusGatewayInstance::teardownGateway()
190 {
191  bool success = true;
192 
193  if (nullptr != m_entireConfig) {
194  json_decref(m_entireConfig);
195  m_entireConfig = nullptr;
196  }
197 
198  if (m_pid != INVALID_PID) {
199  log_debug() << "Killing dbus-proxy with pid " << m_pid;
200 
201  kill(m_pid, SIGKILL); // In some configurations, hangs if using SIGTERM instead
202  waitpid(m_pid, nullptr, 0); // Wait until it exits
203  Glib::spawn_close_pid(m_pid);
204  } else {
205  log_debug() << "dbus-proxy pid not set or already dead: " << m_pid;
206  }
207 
208  if (access(m_socket.c_str(), F_OK) != -1) {
209  if (unlink(m_socket.c_str()) == -1) {
210  log_error() << "Could not remove " << m_socket << ": " << strerror(errno);
211  success = false;
212  }
213  } else {
214  // TODO: Seems weird that this would ever happen. Seems like a severe error.
215  log_debug() << "Socket not accessible, has it been removed already?";
216  }
217 
218  return success;
219 }
